374  /// Output feature model (tree) recursively.
375  ///
376  /// \param[in] Node Root of subtree.
377  /// \param[in] Indent Value to indent statements in dot file.
378  void emitCluster(NodeRef Node, const int Indent = 0) const {
379  O.indent(Indent);
380  emitNode(Node);
381  if (Node->begin() != Node->end()) {
382  O.indent(Indent + 2) << "subgraph cluster_" << static_cast<void *>(Node)
383  << " {\n";
384  O.indent(Indent + 4) << "label=\"\";\n";
385  O.indent(Indent + 4) << "margin=0;\n";
386  O.indent(Indent + 4) << "style=invis;\n";
387  for (auto *Child : *Node) {
388  emitCluster(Child, Indent + 2);
389  O.indent(Indent + 2);
390  if (auto *F = llvm::dyn_cast<vara::feature::Feature>(Child); F) {
391  emitEdge(
392  Node, F,
393  llvm::formatv("arrowhead={0}", F->isOptional() ? "odot" : "dot"));
394  } else {
395  emitEdge(Node, Child, "arrowhead=none");
396  }
397  }
398  O.indent(Indent + 4) << "{\n";
399  O.indent(Indent + 6) << "rank=same;\n";
400  for (auto *Child : *Node) {
401  O.indent(Indent + 6) << "node_" << static_cast<void *>(Child) << ";\n";
402  }
403  O.indent(Indent + 4) << "}\n";
404  O.indent(Indent + 2) << "}\n";
405  }

426  (F->hasLocations() ? llvm::formatv("<hr/><tr><td>{0}</td></tr>",

447  << " margin=.1 fontsize=12 fontname=\"CMU "
